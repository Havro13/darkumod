			}
		}

		available = {
			is_subject = no
			
		}

		visible = {
			OR = {
				original_tag = CZE
				tag = CZE
			}

			has_country_flag = fso_potential_ussa
		}

		complete_effect = {
			set_cosmetic_tag = CZE_USSA
			
			custom_effect_tooltip = form_ussa_cores_tt

				}
			}
			hidden_effect = {
				news_event = { id = fsonews.10 days = 1 }
				set_global_flag = form_ussa_flag
			}
		}

		ai_will_do = {
			factor = 200
		}
